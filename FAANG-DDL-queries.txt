CREATE TABLE "ADMIN"."APPOINTMENTTB" 
   (	"PATIENT_ID" NUMBER, 
	"FNAME" VARCHAR2(255 BYTE), 
	"LNAME" VARCHAR2(255 BYTE), 
	"EMAIL" VARCHAR2(255 BYTE), 
	"CONTACT" VARCHAR2(15 BYTE), 
	"DOCTOR" VARCHAR2(255 BYTE), 
	"PAYMENT" VARCHAR2(255 BYTE)
   );

CREATE OR REPLACE EDITIONABLE TRIGGER "ADMIN"."HOSPITAL_PLACES" 
after insert
    on appointmenttb
begin
    update patients set places=places-1;
end;
/
ALTER TRIGGER "ADMIN"."HOSPITAL_PLACES" ENABLE;
